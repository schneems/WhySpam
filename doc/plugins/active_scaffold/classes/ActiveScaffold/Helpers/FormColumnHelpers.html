active_scaffold_input_for(column, scope = nil)

This method decides which input to use for the given column. It does not do any rendering. It only decides which method is responsible for rendering.

[Source]

    # File vendor/plugins/active_scaffold/lib/active_scaffold/helpers/form_column_helpers.rb, line 7
 7:       def active_scaffold_input_for(column, scope = nil)
 8:         options = active_scaffold_input_options(column, scope)
 9:         # first, check if the dev has created an override for this specific field
10:         if override_form_field?(column)
11:           send(override_form_field(column), @record, options[:name])
12:         # second, check if the dev has specified a valid form_ui for this column
13:         elsif column.form_ui and override_input?(column.form_ui)
14:           send(override_input(column.form_ui), column, options)
15:         # fallback: we get to make the decision
16:         else
17:           if column.association
18:             # if we get here, it's because the column has a form_ui but not one ActiveScaffold knows about.
19:             raise "Unknown form_ui `#{column.form_ui}' for column `#{column.name}'"
20:           elsif column.virtual?
21:             active_scaffold_input_virtual(column, options)
22: 
23:           else # regular model attribute column
24:             # if we (or someone else) have created a custom render option for the column type, use that
25:             if override_input?(column.column.type)
26:               send(override_input(column.column.type), column, options)
27:             # final ultimate fallback: use rails' generic input method
28:             else
29:               # for textual fields we pass different options
30:               text_types = [:text, :string, :integer, :float, :decimal]
31:               options = active_scaffold_input_text_options(options) if text_types.include?(column.column.type)
32:               if column.column.type == :string && options[:maxlength].blank?
33:                 options[:maxlength] = column.column.limit
34:                 options[:size] ||= ActionView::Helpers::InstanceTag::DEFAULT_FIELD_OPTIONS["size"]
35:               end
36:               input(:record, column.name, options.merge(column.options))
37:             end
38:           end
39:         end.to_s + javascript_for_update_column(column, scope, options)
40:       end
