link_to_visibility_toggle(options = {})

Creates a javascript-based link that toggles the visibility of some element on the page. By default, it toggles the visibility of the sibling after the one it‘s nested in. You may pass custom javascript logic in options[:of] to change that, though. For example, you could say :of => ’$("my_div_id")’. You may also flag whether the other element is visible by default or not, and the initial text will adjust accordingly.

132:         url_options = url_options.clone
133:         url_options[:action] = link.action
134:         url_options[:controller] = link.controller if link.controller
135:         url_options.delete(:search) if link.controller and link.controller.to_s != params[:controller]
136:         url_options.merge! link.parameters if link.parameters
137: 
138:         html_options = link.html_options.merge({:class => link.action})
139:         if link.inline?
140:           # NOTE this is in url_options instead of html_options on purpose. the reason is that the client-side
141:           # action link javascript needs to submit the proper method, but the normal html_options[:method]
142:           # argument leaves no way to extract the proper method from the rendered tag.
143:           url_options[:_method] = link.method
144: 
145:           if link.method != :get and respond_to?(:protect_against_forgery?) and protect_against_forgery?
146:             url_options[:authenticity_token] = form_authenticity_token
147:           end
148: 
149:           # robd: protect against submitting get links as forms, since this causes annoying 
150:           # 'Do you wish to resubmit your form?' messages whenever you go back and forwards.
151:         elsif link.method != :get
152:           # Needs to be in html_options to as the adding _method to the url is no longer supported by Rails
153:           html_options[:method] = link.method
154:         end
155: 
156:         html_options[:confirm] = link.confirm if link.confirm?
157:         html_options[:position] = link.position if link.position and link.inline?
158:         html_options[:class] += ' action' if link.inline?
159:         html_options[:popup] = true if link.popup?
160:         html_options[:id] = action_link_id("#{url_options[:parent_controller] + '_' if url_options[:parent_controller]}" + url_options[:action],url_options[:id] || url_options[:parent_id])
161: 
162:         if link.dhtml_confirm?
163:           html_options[:class] += ' action' if !link.inline?
164:           html_options[:page_link] = 'true' if !link.inline?
165:           html_options[:dhtml_confirm] = link.dhtml_confirm.value
166:           html_options[:onclick] = link.dhtml_confirm.onclick_function(controller,action_link_id(url_options[:action],url_options[:id] || url_options[:parent_id]))
167:         end
168:         html_options[:class] += " #{link.html_options[:class]}" unless link.html_options[:class].blank?
169: 
170:         # issue 260, use url_options[:link] if it exists. This prevents DB data from being localized.
171:         label = url_options.delete(:link) || link.label
